#include<iostream>
#include<cmath>
using namespace std;

int main() {
    int n;
    cin >> n;
    int arr[n];
    
    for(int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    int cntNeg = 0, cntZero = 0;
    for(int i = 0; i < n; i++) {
        if(arr[i] == -1) {
            cntNeg++;
        } else if(arr[i] == 0) {
            cntZero++;
        }
    }
    
    if(cntNeg % 2 == 0) {
        cout << cntNeg / 2 + cntZero << endl;
    } else {
        cout << cntNeg / 2 + cntZero + 1 << endl;
    }
    return 0;
}
